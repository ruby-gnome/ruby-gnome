def init
  class << self
    remove_method(:init)
    remove_method(:const_missing)
  end
  loader = Loader.new(self)
  loader.load
  x11_loader = GdkX11::Loader.new(GdkX11)
  x11_loader.load
  if Object.const_defined?(:Gtk) and Gtk.respond_to?(:init)
    Gtk.init
  end

  @@init_hooks.each do |hook|
    hook.call
  end
end
