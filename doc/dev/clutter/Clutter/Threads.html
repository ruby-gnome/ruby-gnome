def add_timeout(interval_or_priority, interval=nil, &block)
  if interval.nil?
    priority, interval = GLib::PRIORITY_DEFAULT, interval_or_priority
    add_timeout_full(priority, interval, &block)
  else
    priority = interval_or_priority
    add_timeout_full(priority, interval, &block)
  end
end
