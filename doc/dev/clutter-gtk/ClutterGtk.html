def init(argv=[])
  class << self
    remove_method(:init)
    remove_method(:const_missing)
  end
  Gtk.init if Gtk.respond_to?(:init)
  loader = Loader.new(self, argv)
  loader.load
  Clutter.init(argv) if Clutter.respond_to?(:init)
end
