def open_stream(path, flags=nil)
  flags ||= ResourceLookupFlags::NONE
  input_stream = open_stream_raw(path, flags)
  if block_given?
    begin
      yield(input_stream)
    ensure
      input_stream.close
    end
  else
    input_stream
  end
end
