def initialize(*args)
  if args.size == 1
    case args[0]
    when :any
      initialize_new_any
      return
    when :empty
      initialize_new_empty
      return
    end
  end
  initialize_raw(*args)
end

def []=(name, type=nil, value)
  if type
    value_type = GLib::Type.const_get(type.to_s.upcase)
    set_value(name, GLib::Value.new(value_type, value))
  else
    set_value(name, value)
  end
end
