def post_load(repository, namespace)
  @base_module.constants.each do |constant|
    case constant
    when :INVOKERS,
         :Loader
      next
    else
      if constant.to_s.upcase == constant.to_s
        name = "FT2_#{constant}"
      else
        name = "FT2#{constant}"
      end
      value = @base_module.const_get(constant)
      Pango.const_set(name, value)
    end
  end
end
