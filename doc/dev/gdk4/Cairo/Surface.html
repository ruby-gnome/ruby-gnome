def to_pixbuf(options={})
  src_x = options[:src_x] || 0
  src_y = options[:src_y] || 0
  w = options[:width]
  h = options[:height]
  if w.nil? and respond_to?(:width)
    w = width
  end
  if h.nil? and respond_to?(:height)
    h = height
  end
  to_pixbuf_raw(src_x, src_y, w, h)
end
