def initialize(options={})
  entry = options[:entry]
  model = options[:model]

  if entry
    if model
      initialize_new_with_model_and_entry(model)
    else
      initialize_new_with_entry
    end
  else
    if model
      initialize_new_with_model(model)
    else
      initialize_new
    end
  end
end

def active_iter

  if found
    iter.model = model
    iter
  else
    nil
  end
end
