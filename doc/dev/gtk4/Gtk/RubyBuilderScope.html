def virtual_do_create_closure(builder,
                              function_name,
                              flags,
                              object)
  object ||= builder.current_object || builder
  method_name = normalize_name(function_name)
  GLib::Closure.new do |*args|
    object.__send__(method_name, *args)
  end
end
